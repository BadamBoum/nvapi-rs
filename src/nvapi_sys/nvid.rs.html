use std::mem;

macro_rules! nvapis {
    ($(
        $name:ident = $id:expr,
    )*) => {
        #[repr(u32)]
        #[derive(Debug, Copy, Clone, PartialOrd, Ord, PartialEq, Eq, Hash)]
        pub enum Api {
        $(
            $name = $id,
        )*
        }

        impl Api {
            pub fn from_id(id: u32) -> Result<Self, ()> {
                match id {
                $(
                    $id
                )|* => Ok(unsafe { mem::transmute(id) }),
                    _ => Err(()),
                }
            }

            pub fn id(&self) -> u32 {
                *self as _
            }
        }
    };
}
