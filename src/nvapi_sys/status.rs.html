impl error::Error for Status {
    fn description(&self) -> &str {
        "NVAPI Error"
    }
}

impl fmt::Display for Status {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        fmt::Debug::fmt(self, f)
    }
}
